## Overview

`ExtendedClient` is the foundation of the Poddy framework. It extends the Discord.js `Client` class to provide:

- Automatic command, button, modal, and select menu handling
- Database access via Prisma ORM
- i18n language support
- Caching layers for guilds, roles, and members
- DataDog metrics integration

## Creating a custom client

You can extend `ExtendedClient` to add custom functionality:

```typescript lib/extensions/ExtendedClient.ts
import ExtendedClient from "@lib/extensions/ExtendedClient.js";
import CustomFunctions from "@src/utilities/functions.js";

export class MyClient extends ExtendedClient {
  override get functions(): CustomFunctions {
    return new CustomFunctions(this);
  }
}
```

```typescript src/index.ts
import { MyClient } from "@src/client.js";
import { REST } from "@discordjs/rest";
import { WebSocketManager } from "@discordjs/ws";

const client = new MyClient({
  rest: new REST({ version: "10" }).setToken(process.env.DISCORD_TOKEN),
  gateway: new WebSocketManager({
    token: process.env.DISCORD_TOKEN,
    intents: GatewayIntentBits.Guilds | GatewayIntentBits.GuildMessages,
  }),
  options: {
    // Your custom options
  },
});

await client.start();
```

## Example usage

```typescript src/bot/events/ready.ts
import EventHandler from "@lib/classes/EventHandler.js";
import type ExtendedClient from "@lib/extensions/ExtendedClient.js";
import { GatewayDispatchEvents } from "discord-api-types/v10";

export default class Ready extends EventHandler {
  public constructor(client: ExtendedClient) {
    super(client, GatewayDispatchEvents.Ready, true);
  }

  public override async run({ data, shardId }) {
    this.client.logger.info(
      `Logged in as ${data.user.username} on shard ${shardId}`
    );

    // Access all client features
    const guilds = await this.client.prisma.guild.findMany();
    this.client.logger.info(`Serving ${guilds.length} guilds`);
  }
}
```
